79int Warehouse::sortAndCount(DynArray<Product*> products) const {
80 if (products.size() == 0) {
81 return 0;
82 }
83
84 int quantitySum = 0;
85 for (int i = 0; i < products.size() - 1; i++) {
86 int minIndex = i;
87 for (int j = i + 1; j < products.size(); j++) {
88 if (products[j] < products[minIndex]) {
89 minIndex = j;
90 }
91 }
92 products.swap(i, minIndex);
93 quantitySum += products[i]->getQuantity();
94 }

235bool Warehouse::takeOutProduct(std::ostream& out, std::istream& in, char const* name, int wanted) {
236 DynArray<Product*> products(10);
237 findAllByName(name, products);
238 int productCount = sortAndCount(products);
239
240 if (productCount < wanted && insufficientQuantity(out, in, productCount, products)) {
241 return false;
242 }
243

245 Placement p = products[i]->getPlacement();
246 int currentQuantity = products[i]->getQuantity();

281bool Warehouse::restock(Product const& product) {
282 Product* found = findEqual(product.getName(), product.getExpirationDate());
283
284 if (found) {
285 Placement pNew, p = found->getPlacement();
286
287 // Place next to item on the shelf
288 if (addProduct(product, p.section, p.shelf, p.index - 1)) {
289 return true;
290 }
291 if (addProduct(product, p.section, p.shelf, p.index + 1)) {
292 return true;
293 }
294
295 // Place on the same shelf
296 pNew = sections[p.section, p.shelf].addProduct(product);
297 if (pNew.index != -1) {
298 sections[pNew.section][pNew.shelf][pNew.index].setPlacement(pNew);
299 return true;
300 }
301
302 // Place on close by shelves in the same section
303 pNew = sections[p.section, p.shelf + 1].addProduct(product);
304 if (pNew.index != -1) {
305 sections[pNew.section][pNew.shelf][pNew.index].setPlacement(pNew);
306 return true;

309 if (pNew.index != -1) {
310 sections[pNew.section][pNew.shelf][pNew.index].setPlacement(pNew);
311 return true;
312 }
313
314 // Place in close by sections
